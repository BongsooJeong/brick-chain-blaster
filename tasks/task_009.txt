```dart
class BallProgressionSystem {
  int destroyedBricks = 0;
  int ballCount = 1;
  int nextThreshold = 10; // Initial threshold
  
  void onBrickDestroyed() {
    destroyedBricks++;
    checkForBallIncrease();
  }
  
  
  void checkForBallIncrease() {
    if (destroyedBricks >= nextThreshold) {
      increaseBallCount();
      calculateNextThreshold();
    }
  }
  
  void increaseBallCount() {
    ballCount++;
    showBallIncreaseAnimation();
  }
  
  void calculateNextThreshold() {
    // Exponential or linear increase in threshold
    nextThreshold = (nextThreshold * 1.5).round();
  }
  
  double getProgressToNextBall() {
    return destroyedBricks / nextThreshold;
  }
}
```
